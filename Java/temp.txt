class MinSizeSubArraySum {
  public static int findMinSubArray(int S, int[] arr) {
    // TODO: Write your code here

    //pre condition
    //normal condition
    int smallestLength = Integer.MAX_VALUE;
    int left_cursor = 0;
    int current_sum = 0;

    for(int right_cursor = 0; right_cursor < arr.length; right_cursor++)
    {
        current_sum += arr[right_cursor];

        //trim window
        while(current_sum >= S)
        {
            //window valid
            smallestLength = Math.min( right_cursor - left_cursor + 1, smallestLength);

            //prepare for next window
            current_sum -= arr[left_cursor];
            left_cursor++;
        }
    }
    return smallestLength;
  }
}
